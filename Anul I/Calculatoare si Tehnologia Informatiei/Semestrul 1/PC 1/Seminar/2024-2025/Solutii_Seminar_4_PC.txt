#include <stdio.h>
#include <stdlib.h>

void min_max(int n, int *mini, int *maxi) {
    *maxi=0;
    *mini=9;
    while (n) 
    {
        if (n % 10 < *mini) *mini = n % 10;
        if (n % 10 > *maxi) *maxi = n % 10;
        n /= 10;
    }
}

int main()
{
    int n;
    scanf("%d", &n);
    FILE *g=fopen("date.out","w");
    for (int x = 1; x <= n; x++) 
    {
        int mini,maxi;
        min_max(x, &mini, &maxi);
        if (maxi == mini) 
            fprintf(g,"%d ", x);
    }
    fclose(g);
return 0;
}

int rearanjare (int n, int v[])
{
    int cnt=0;
    for (int i=1; i<=n-cnt; ++i) {
        if (v[i]==0) {
            cnt++;
            for (int j=i+1; j<=n; ++j) v[j-1]=v[j];
            v[n]=0;
            --i;
        }
    }
    return cnt;
}

int n,a[10][10]; 
void matrice(int n) 
{ 
    int i,j; 
    for(i=0;i<n;i++) 
    { 
        for(j=0;j<n;j++) 
        { 
            if(i>j) a[i][j]=i-j; 
            else a[i][j]=j-i; 
        } 
    } 
}

{ 
    int i,j;

    f=fopen("matrice.txt","w"); 
    for(i=0;i<n;i++)

}
