    a, b = b, a

Problema propus #4

    case "+":
        print(f"{x} {op} {y} = {x + y}")
    case "-":
        print(f"{x} {op} {y} = {x - y}")
    case "*":
        print(f"{x} {op} {y} = {x * y}")

        print(f"{x} {op} {y} = {x / y}")
    case "**":
        print(f"{x} {op} {y} = {x ** y}")
    case _:
        print("Acesta nu este un operator!")
